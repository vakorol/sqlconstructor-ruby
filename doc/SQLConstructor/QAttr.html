def initialize ( init_hash = nil )
    if init_hash.is_a? Hash
        @name           = init_hash[:name]
        @text           = init_hash[:text]
        @val            = init_hash[:val]
        @val_type       = init_hash[:val_type]

def to_s
    if [ SQLValList, SQLAliasedList ].include? @val
        result = @val.to_s
    else 
        result = @text
        if @val
            val_arr = @val.is_a?( Array )  ? @val  : [ @val ]
            result += " " + val_arr.join( "," )
        end
    end
    return result
end
